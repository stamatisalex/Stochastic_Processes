# random walk generator:
def rand_walk_Z(start = 0, probs = (0.5, 0.5)):
    x = start
    steps = [-1, 1]  # probs order should be (left, right)
    while True:  # continue for ever in needed
        x += choice(steps, p=probs)
        yield x  # it's a generator!

# random walk in Z^2 generator:
def rand_walk_Z2(start=0, probs=[1/3,2/3]):
    x = start
    while True:  
        if (x == start):
            x += np.random.choice([1, 0], p=probs)
        else:
            x += np.random.choice([1,-1], p=probs)
        yield x

np.random.seed(2021)  # for reproducibility
state = [[ 0, 1 ], [ 0, -1 ], [ 1, 0 ], [ -1, 0 ] ]
list1=[]
means = []
M=500
rad=range( 10, 101, 10)
for R in rad:
    count=[]
    for i in range(M):
        time=0
        x = 0
        y=0
        while(x**2 + y ** 2 <= R ** 2 ):
            check = state[np.random.randint( 4 )]
            x += check[ 0 ]
            y += check[ 1 ]
            time += 1
        count.append(time)
    list1.append(R)
    mean_value= stat.mean(count)
    means.append(mean_value)

fig, ax = plt.subplots()  # initiate plots 
plt.loglog(list1,means,base=2)
plt.title( "Average Escape Time at given radius")
plt.xlabel( "Radius" )
