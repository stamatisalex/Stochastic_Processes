import random
import statistics as stat
import matplotlib.pyplot as plt 
import numpy as np
from math import log

%matplotlib inline

def Vol1(d):
    x = d/2
    return pi ** x / gamma(x + 1)

l1 = []
for i in range(2,21):
    l1.append(Vol1(i)/2**i)

logp = [np.log(x) for x in l1]
d = np.arange(2,21)
plt.figure(figsize=(12, 7))
plt.plot(d,d-logp);
plt.xlabel("Dimension");
plt.ylabel("d-Log(p(d))");

Παραπάνω σχεδιάστηκαν και η γραφική παράσταση "d-logp(d)" και η "logp(d)". Παρατηρούμε ότι το σχετικό σφάλμα είναι μεγάλο για τις μεγάλες διαστάσεις. Ερώτημα γ¶

γ) Αν θέλατε να πάρετε 1000 δείγματα στη μοναδιαία μπάλα 20 διαστάσεων με τη μέθοδο της απόρριψης όπως παραπάνω, πόσα περίπου δείγματα στον κύβο θα έπρεπε να φτιάξετε? Κάντε μια εκτίμηση του χρόνου που θα χρειαζόταν αυτό στον υπολογιστή σας.

Λύση

Η πιθανότητα να πέσει ένα τυχαίο σημείο μέσα στην μπαλα 20 διαστάσεων είναι $\frac{\omega(20)}{2^{20}} = \frac{\frac{\pi^{10}}{\Gamma(11)}}{2^{20}} = \frac{\pi^{10}}{2^{20}\cdot \Gamma(11)}= \frac{1}{4\cdot 10^{7}}$. Άρα για ένα μόνο δείγμα επιτυχίας χρειάζονται $40.000.000$ προσπάθειες

Άρα για 1000 τέτοια δείγματα θα χρειαστούν περίπου $4 \cdot 10^{10}$ επαναλήψεις στον υπολογιστή μου. Δεδομένου ότι θέλω περίπου $4 sec$ για $1.000.000$ πράξεις στον υπολογιστή μου, τότε θα χρειαστούν $ 4 \cdot 10^4 sec = 11\ hours$.

import matplotlib.pyplot as plt
import random as r

for delta in[1.0 ,0.01,20.0]:
    N = 100 # number of steps to approach equilibrium
    samples = 1000 
    point_x = []
    point_y = []

    for _ in range(samples):
        x = [0,0]  ## start at the centre of th disc. This variable will keep the position of the chain
        R_sq = 0.0  ## this variable keeps the squared distance from 0. It saves some computations to keep it

        for _ in range(N):
            k = r.choice([0,1])  ## choose a jump direction at random
            z = r.uniform(-delta,delta) ## choose a jump size uniformly in (-delta,delta)
            x_prop_k = x[k] + z   ## propose a jump by z in the direction k
            R_sqprop = R_sq - x[k]**2+ x_prop_k**2 ## compute the squared distance from 0 after the proposed jump 
            if R_sqprop < 1.0: 
                R_sq = R_sqprop
                x[k]= x_prop_k   ## if the proposed jump leads to a point in the disc, then jump

        point_x.append(x[0])
        point_y.append(x[1])

    plt.gca().set_aspect('equal', adjustable='box')
    plt.title('For delta ={}'.format(delta))
    plt.rcParams['figure.figsize'] = [6,6];
    plt.scatter(point_x,point_y);
    plt.show();

Παραπάνω βλέπουμε τα 3 διαγράμματα για delta=1 , delta =0.01 και delta =20 αντίστοιχα.Στην περίπτωση που delta =0.01 τα δείγματα δεν "προλαβαίνουν" να απομακρινθούν σε 100 βηματα αρκεά από το κέντρο, αφού όπως βλέπουμε ελάχιστα ξεπερνούν την απόσταση του 0.1 από το (0,0). Αντιθέτως, στην δεύτερη περίπτωση οι κινήσεις πολλών δειγμάτων απορρίπτονται με αποτέλεσμα τα δείγματα αυτά να μην φεύγουν από τους άξονες x ή y και να συγκεντρώνονται σε αυτούς οπως βλέπουμε.

Γράψτε μια ρουτίνα που θα τυπώνει για διαστάσεις $d=2,3,...,100$ τον εκτιμόμενο με MCMC όγκo της μοναδιαίας μπάλας σε $d$ διαστάσεις, την πραγματική τιμή του και το σχετικό σφάλμα της εκτίμησης. Μπορείτε να ξεκινήσετε από την προφανή σχέση $|D_1|=2$ και να δουλέψετε επαγωγικά για να υπολογίσετε τα ω(2), ω(3),... Στο jupyter notebook που θα παραδώσετε τρέξτε τη ρουτίνα σας ώστε να φαίνονται τα αποτελέσματα.

Λύση

Ξέρωντας ότι είναι $|D_1| = 2$, οπότε μέσω της MCMC δειγματοληψίας εκτιμούμε τον λόγο $\frac{|D_{d+1}|}{|C_{d+1}|}$= $\frac{|D_{d+1}|}{2|D_d|}$ που είναι ο λόγος των σημείων του κυλίνδρου σε d+1 διαστάσεις που πέφτουν μέσα στην d+1-διάστατη σφαίρα. Έτσι εκτιμούμε το $|D_{d+1}|$ ενώ ξέρουμε το $|D_d|$. Ο κώδικας της άσκησης είναι ο εξής:

import matplotlib.pyplot as plt
import random as r
import numpy as np


delta = 1
N = 100
lastd = 2
samples =10000     
for d in range(1,30): 
    samples_inside_D = 0 

    for _ in range(samples):
        x=[0]*d
        R_sq = 0 

        for _ in range(N): 
            k = r.choice(np.arange(d))
            z = r.uniform(-delta,delta)
            x_prop_k = x[k] +z
            R_sqprop = R_sq - x[k]**2 + x_prop_k**2
            if R_sqprop < 1.0: 
                R_sq = R_sqprop
                x[k]= x_prop_k

        x_dplus1 = r.uniform(-1,1)
        R_sq += x_dplus1**2

        if R_sq<1: 
            samples_inside_D += 1


    D=2*lastd*(samples_inside_D/samples)
    lastd=D
    error= abs(Vol1(d+1)-D)/Vol1(d+1)
    print("The Monte Carlo estimate of ω(", d+1 ,") is:\t%.5f " % D)
    print("The actual value of ω(", d+1 ,") is:\t\t%.5f " % Vol1(d+1))
    print("The relative error is:\t\t\t%.5f" % error)
    print("\n")
